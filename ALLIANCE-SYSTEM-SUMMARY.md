  id TEXT PRIMARY KEY,
  name TEXT UNIQUE,
  tag TEXT UNIQUE,
  description TEXT,
  leader_id TEXT,
  max_members INTEGER DEFAULT 50,

  created_at INTEGER,
  expires_at INTEGER,

);

  id TEXT PRIMARY KEY,
  alliance_id TEXT,
  sender_id TEXT,
  content TEXT,
  created_at INTEGER,
  edited_at INTEGER,
  deleted_at INTEGER
);
